33  package org.beiter.michael.db;
34  
35  import org.apache.commons.lang3.Validate;
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  
39  import java.sql.Connection;
40  import java.sql.SQLException;
41  
42  /**
43   * This class creates and manages JDBC Connection instances either from:
44   * <ul>
45   * <li>a named JNDI managed connection or</li>
46   * <li>a connection pool that is maintained by the {@link DataSourceFactory} factory</li>
47   * </ul>
48   */
49  public final class ConnectionFactory {
50  
51      /**
52       * The logger object for this class
53       */
54      private static final Logger LOG = LoggerFactory.getLogger(ConnectionFactory.class);
55  
56      /**
57       * A private constructor to prevent instantiation of this class
58       */
59      private ConnectionFactory() {
60      }
61  
62      /**
63       * Return a Connection instance for a JNDI managed JDBC connection.
64       *
65       * @param jndiName The JNDI connection name
66       * @return a JDBC connection
67       * @throws FactoryException         When the connection cannot be retrieved from JNDI
68       * @throws NullPointerException     When {@code jndiName} is null
69       * @throws IllegalArgumentException When {@code jndiName} is empty
70       */
71      public static Connection getConnection(final String jndiName)
72              throws FactoryException {
73  
74          Validate.notBlank(jndiName, "The validated character sequence 'jndiName' is null or empty");
75  
76          // no need for defensive copies of Strings
77  
78          try {
79              return DataSourceFactory.getDataSource(jndiName).getConnection();
80          } catch (SQLException e) {
81              final String error = "Error retrieving JDBC connection from JNDI: " + jndiName;
82              LOG.warn(error);
83              throw new FactoryException(error, e);
84          }
85      }
86  
87      /**
88       * Return a Connection instance from a pool that manages JDBC driver based connections.
89       * <p>
90       * The driver-based connection are managed in a connection pool. The pool is created using the provided properties
91       * for both the connection and the pool spec. Once the pool has been created, it is cached (based on URL and
92       * username), and can no longer be changed. Subsequent calls to this method will return a connection from the
93       * cached pool, and changes in the pool spec (e.g. changes to the size of the pool) will be ignored.
94       *
95       * @param poolSpec A connection pool spec that has the driver and url configured as non-empty strings
96       * @return a JDBC connection
97       * @throws FactoryException         When the connection cannot be retrieved from the pool, or the pool cannot be
98       *                                  created
99       * @throws NullPointerException     When the {@code poolSpec}, {@code poolSpec.getDriver()}, or
100      *                                  {@code poolSpec.getUrl()} are {@code null}
101      * @throws IllegalArgumentException When {@code poolSpec.getDriver()} or {@code poolSpec.getUrl()} are empty
102      */
103     public static Connection getConnection(final ConnectionProperties poolSpec)
104             throws FactoryException {
105 
106         Validate.notNull(poolSpec, "The validated object 'poolSpec' is null");
107         Validate.notBlank(poolSpec.getDriver(),
108                 "The validated character sequence 'poolSpec.getDriver()' is null or empty");
109         Validate.notBlank(poolSpec.getUrl(), "The validated character sequence 'poolSpec.getUrl()' is null or empty");
110 
111         // no need for defensive copies of Strings
112 
113         try {
114             return DataSourceFactory.getDataSource(poolSpec).getConnection();
115         } catch (SQLException e) {
116             // a connection is identified by the URL, the username, and the password
117             final String key = String.format("%s:%s", poolSpec.getUrl(), poolSpec.getUsername());
118             final String error = "Error retrieving JDBC connection from pool: " + key;
119             LOG.warn(error);
120             throw new FactoryException(error, e);
121         }
122     }
123 
124     /**
125      * Resets the internal state of the {@link DataSourceFactory} that manages the data source pools exposed by this
126      * factory.
127      * <p>
128      * <strong>This method does not release any resources that have been borrowed from the connection pools managed
129      * by this factory.</strong> To avoid resource leaks, you <strong>must</strong> close / return all connections to
130      * their pools before calling this method.
131      */
132     public static void reset() {
133 
134         // Unset the cached connections
135         DataSourceFactory.reset();
136     }
137 }
